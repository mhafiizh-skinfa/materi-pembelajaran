Step 8.1 — Error Handling

Menangani error agar program tidak crash

📌 Jenis Error di JavaScript

  • SyntaxError    → kesalahan penulisan kode
  • ReferenceError → variabel tidak ditemukan
  • TypeError     → tipe data tidak sesuai
  • RangeError    → nilai di luar batas yang diizinkan
  • CustomError   → error buatan sendiri
📄 Try / Catch / Finally 
// Struktur dasar
try {
  // kode yang mungkin error
  throw new Error("Terjadi kesalahan!");
} catch (e) {
  console.log(e.name);    // Error
  console.log(e.message); // Terjadi kesalahan!
  console.log(e.stack);   // stack trace lengkap
} finally {
  console.log("Selalu dijalankan"); // meskipun ada error
}


// Contoh nyata: validasi input
function bagi(a, b) {
  if (typeof a !== "number" || typeof b !== "number") {
    throw new TypeError("Input harus berupa angka!");
  }
  if (b === 0) {
    throw new RangeError("Tidak bisa bagi dengan 0!");
  }
  return a / b;
}

try {
  console.log(bagi(10, 2));   // ✅ 5
  console.log(bagi(10, 0));   // ❌ RangeError
} catch (e) {
  if (e instanceof TypeError) {
    console.log("❌ Tipe salah:", e.message);
  } else if (e instanceof RangeError) {
    console.log("❌ Range salah:", e.message);
  } else {
    console.log("❌ Error lain:", e.message);
  }
}
📄 Custom Error Class 
// Buat class Error sendiri
class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name  = "ValidationError";
    this.field = field;
  }
}

class NotFoundError extends Error {
  constructor(resource, id) {
    super(`${resource} dengan id ${id} tidak ditemukan`);
    this.name     = "NotFoundError";
    this.resource = resource;
    this.id       = id;
  }
}

function validasiSiswa(siswa) {
  if (!siswa.nama) {
    throw new ValidationError("Nama wajib diisi!", "nama");
  }
  if (siswa.nilai < 0 || siswa.nilai > 100) {
    throw new ValidationError("Nilai harus antara 0-100!", "nilai");
  }
  return "✅ Data valid!";
}

try {
  validasiSiswa({ nama: "", nilai: 85 });
} catch (e) {
  if (e instanceof ValidationError) {
    console.log(`❌ [${e.field}] ${e.message}`);
    // ❌ [nama] Nama wajib diisi!
  }
}
Step 8.2 — ES Module

📌 Konsep

  • Module = memecah kode menjadi file terpisah agar lebih terorganisir
  • export → kirim keluar dari file
  • import → ambil dari file lain
  • Gunakan type="module" di tag script HTML
Jenis Sintaks Export Sintaks Import
Named Export export const x = ... import { x } from "./file"
Default Export export default function import namaApa from "./file"
Import All export { a, b, c } import * as Nama from "./file"
Rename export { x as y } import { y as z } from "./file"
📄 math.js — File Module 
// Named export
export const PI = 3.14159265;

export function tambah(a, b)   { return a + b; }
export function kurang(a, b)   { return a - b; }
export function kali(a, b)     { return a * b; }
export function bagi(a, b) {
  if (b === 0) throw new Error("Tidak bisa bagi 0!");
  return a / b;
}
export function pangkat(base, exp) { return base ** exp; }

// Default export
export default function kalkulator(a, op, b) {
  switch(op) {
    case "+": return tambah(a, b);
    case "-": return kurang(a, b);
    case "*": return kali(a, b);
    case "/": return bagi(a, b);
    default: throw new Error("Operator tidak dikenal!");
  }
}
📄 app.js — File Utama 
// Import default
import kalkulator from "./math.js";

// Import named
import { PI, tambah, pangkat } from "./math.js";

// Import semua
import * as Math from "./math.js";

// Import dengan rename
import { tambah as add, kurang as subtract } from "./math.js";

console.log(PI);                    // 3.14159265
console.log(tambah(5, 3));          // 8
console.log(pangkat(2, 10));        // 1024
console.log(kalkulator(10, "*", 5)); // 50
console.log(Math.bagi(20, 4));      // 5

// Dynamic import (lazy loading)
async function muatModule() {
  const { tambah } = await import("./math.js");
  console.log(tambah(3, 4)); // 7
}
Step 8.3 — Iterator & Generator

📌 Konsep

  • Iterator → object yang punya method next() untuk iterasi satu per satu
  • Generator → function khusus yang bisa dijeda dan dilanjutkan
  • Tandai dengan function* dan gunakan yield untuk menghasilkan nilai
📄 Iterator 
// Membuat iterator manual
function buatIterator(arr) {
  let index = 0;
  return {
    next() {
      return index < arr.length
        ? { value: arr[index++], done: false }
        : { value: undefined,   done: true  };
    }
  };
}

const iter = buatIterator(["Apel", "Mangga", "Jeruk"]);
console.log(iter.next()); // { value: "Apel",   done: false }
console.log(iter.next()); // { value: "Mangga", done: false }
console.log(iter.next()); // { value: "Jeruk",  done: false }
console.log(iter.next()); // { value: undefined, done: true }
📄 Generator Function 
// Generator → function* + yield
function* hitungMundur(n) {
  while (n > 0) {
    yield n--; // pause di sini, kirim nilai, lanjut nanti
  }
}

const gen = hitungMundur(3);
console.log(gen.next()); // { value: 3, done: false }
console.log(gen.next()); // { value: 2, done: false }
console.log(gen.next()); // { value: 1, done: false }
console.log(gen.next()); // { value: undefined, done: true }

// Generator bisa dipakai di for...of
for (let n of hitungMundur(5)) {
  console.log(n); // 5, 4, 3, 2, 1
}

// Generator infinite (tak terbatas)
function* idGenerator() {
  let id = 1;
  while (true) {
    yield id++;
  }
}

const buatId = idGenerator();
console.log(buatId.next().value); // 1
console.log(buatId.next().value); // 2
console.log(buatId.next().value); // 3

// Generator dengan yield*
function* gabung(arr1, arr2) {
  yield* arr1; // delegasikan ke array lain
  yield* arr2;
}

const hasil = [...gabung([1,2,3], [4,5,6])];
console.log(hasil); // [1, 2, 3, 4, 5, 6]
Step 8.4 — Proxy & Reflect

📌 Konsep

  • Proxy → membuat "perantara" untuk object, bisa intercept operasi (get, set, dll)
  • Reflect → API standar untuk operasi object (pasangan dari Proxy)
  • Gunakan untuk: validasi, logging, caching, access control
📄 Proxy Dasar 
// Proxy dengan handler get & set
const siswa = { nama: "Budi", nilai: 85, umur: 17 };

const handler = {
  // Intercept saat READ property
  get(target, key) {
    console.log(`📖 Mengakses: ${key}`);
    return Reflect.get(target, key);
  },

  // Intercept saat WRITE property
  set(target, key, value) {
    if (key === "nilai" && (value < 0 || value > 100)) {
      throw new RangeError("Nilai harus antara 0-100!");
    }
    console.log(`✏️ Mengubah ${key}: ${target[key]} → ${value}`);
    return Reflect.set(target, key, value);
  },

  // Intercept saat DELETE property
  deleteProperty(target, key) {
    console.log(`🗑️ Menghapus: ${key}`);
    return Reflect.deleteProperty(target, key);
  }
};

const siswaProxy = new Proxy(siswa, handler);

console.log(siswaProxy.nama);  // 📖 Mengakses: nama → Budi
siswaProxy.nilai = 95;          // ✏️ Mengubah nilai: 85 → 95
siswaProxy.nilai = 150;         // ❌ RangeError: Nilai harus 0-100!
delete siswaProxy.umur;         // 🗑️ Menghapus: umur
📄 Proxy untuk Validasi Form 
// Contoh nyata: validasi form otomatis dengan Proxy
function buatFormValidator(schema) {
  const data = {};

  return new Proxy(data, {
    set(target, field, value) {
      const rules = schema[field];
      if (!rules) throw new Error(`Field "${field}" tidak diizinkan!`);

      if (rules.required && !value) {
        throw new Error(`"${field}" wajib diisi!`);
      }
      if (rules.minLength && value.length < rules.minLength) {
        throw new Error(`"${field}" minimal ${rules.minLength} karakter!`);
      }
      if (rules.type && typeof value !== rules.type) {
        throw new TypeError(`"${field}" harus bertipe ${rules.type}!`);
      }

      return Reflect.set(target, field, value);
    }
  });
}

const formSiswa = buatFormValidator({
  nama : { required: true, minLength: 3, type: "string" },
  email: { required: true, type: "string" },
  nilai: { required: true, type: "number" },
});

try {
  formSiswa.nama  = "Budi Santoso"; // ✅
  formSiswa.email = "budi@email.com"; // ✅
  formSiswa.nilai = "sembilan puluh"; // ❌ TypeError
} catch (e) {
  console.log("❌", e.message);
  // ❌ "nilai" harus bertipe number!
}
