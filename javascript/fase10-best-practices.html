Step 10.1 — Clean Code

📌 Prinsip Clean Code

// ===== 1. DEBOUNCE — tunda eksekusi sampai user berhenti ketik =====
function debounce(func, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => func.apply(this, args), delay);
  };
}

const cariDebounced = debounce((keyword) => {
  console.log("Mencari:", keyword); // hanya dipanggil setelah 500ms berhenti
}, 500);

document.querySelector("#search")
  .addEventListener("input", (e) => cariDebounced(e.target.value));


// ===== 2. THROTTLE — batasi eksekusi maksimal 1x per interval =====
function throttle(func, limit) {
  let inThrottle;
  return function(...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

const handleScrollThrottled = throttle(() => {
  console.log("Scroll!", window.scrollY); // max 1x per 200ms
}, 200);

window.addEventListener("scroll", handleScrollThrottled);


// ===== 3. MEMOIZE — cache hasil komputasi berat =====
function memoize(func) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      console.log("📦 Dari cache");
      return cache.get(key);
    }
    const result = func.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

const faktorialMemo = memoize(function faktorial(n) {
  return n <= 1 ? 1 : n * faktorial(n - 1);
});

console.log(faktorialMemo(10)); // hitung
console.log(faktorialMemo(10)); // 📦 Dari cache — langsung!


// ===== 4. LAZY LOADING — muat konten hanya saat dibutuhkan =====
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target;
      img.src = img.dataset.src; // load gambar saat terlihat
      observer.unobserve(img);
    }
  });
});

document.querySelectorAll("img[data-src]")
  .forEach(img => observer.observe(img));
Step 10.3 — Checklist Best Practices

✅ Checklist Sebelum Submit Kode

✓
Gunakan const untuk nilai tetap, let untuk yang berubah. Hindari var
✓
Nama variabel / function deskriptif dan mengikuti konvensi camelCase / PascalCase
✓
Setiap function punya satu tanggung jawab (Single Responsibility Principle)
✓
Gunakan async/await dengan try/catch untuk semua operasi async
✓
Validasi semua input dari user sebelum diproses
✓
Hindari magic number — simpan di konstanta bernama
!
Gunakan event delegation untuk elemen yang dibuat secara dinamis
!
Jangan lupa removeEventListener jika elemen sudah tidak dipakai
!
Hindari innerHTML dengan data dari user (gunakan textContent atau sanitasi)
✗
Jangan simpan data sensitif (password, token) di localStorage
✗
Jangan gunakan eval() — sangat berbahaya!
✗
Jangan biarkan console.log() di kode produksi

📄 app.js — Kode Utama 
// ===== KONSTANTA =====
const STORAGE_KEY  = "dataSiswa";
const NILAI_LULUS  = 75;

// ===== CLASS SISWA =====
class Siswa {
  constructor(nama, kelas, nilai) {
    this.id    = Date.now();
    this.nama  = nama.trim();
    this.kelas = kelas.trim();
    this.nilai = Number(nilai);
  }

  get grade() {
    const g = [
      { min: 90, val: "A" }, { min: 80, val: "B" },
      { min: 70, val: "C" }, { min: 60, val: "D" },
      { min: 0,  val: "E" },
    ];
    return g.find(x => this.nilai >= x.min).val;
  }

  get isLulus() {
    return this.nilai >= NILAI_LULUS;
  }
}

// ===== STORAGE SERVICE =====
const StorageService = {
  ambil()        { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; },
  simpan(data)   { localStorage.setItem(STORAGE_KEY, JSON.stringify(data)); },
  bersihkan()   { localStorage.removeItem(STORAGE_KEY); }
};

// ===== VALIDASI =====
const Validasi = {
  nama(nama) {
    if (!nama || nama.trim().length < 3)
      throw new Error("Nama minimal 3 karakter!");
  },
  kelas(kelas) {
    if (!kelas || kelas.trim() === "")
      throw new Error("Kelas wajib diisi!");
  },
  nilai(nilai) {
    const n = Number(nilai);
    if (isNaN(n) || n < 0 || n > 100)
      throw new Error("Nilai harus antara 0 - 100!");
  }
};

// ===== STATE =====
let dataSiswa   = StorageService.ambil();
let filterNama  = "";
let filterKelas = "";
let sortBy      = "nama";

// ===== HELPER =====
function debounce(func, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => func(...args), delay);
  };
}

function tampilError(pesan) {
  const el = document.getElementById("errorMsg");
  el.textContent = pesan;
  el.classList.remove("hidden");
  setTimeout(() => el.classList.add("hidden"), 3000);
}

function getDataTerfilter() {
  return dataSiswa
    .filter(s => s.nama.toLowerCase().includes(filterNama.toLowerCase()))
    .filter(s => filterKelas ? s.kelas === filterKelas : true)
    .sort((a, b) =>
      sortBy === "nilai"
        ? b.nilai - a.nilai
        : a.nama.localeCompare(b.nama)
    );
}

// ===== RENDER =====
function renderSummary() {
  const total    = dataSiswa.length;
  const lulus    = dataSiswa.filter(s => s.nilai >= NILAI_LULUS).length;
  const rataRata = total
    ? (dataSiswa.reduce((acc, s) => acc + s.nilai, 0) / total).toFixed(1)
    : 0;

  document.getElementById("summary").innerHTML = `
    <span>👥 Total: <b>${total}</b></span>
    <span>✅ Lulus: <b>${lulus}</b></span>
    <span>❌ Tidak: <b>${total - lulus}</b></span>
    <span>📊 Rata²: <b>${rataRata}</b></span>
  `;
}

function renderTabel() {
  const data    = getDataTerfilter();
  const tbody   = document.getElementById("bodyTabel");
  const emptyEl = document.getElementById("emptyMsg");

  tbody.innerHTML = "";

  if (data.length === 0) {
    emptyEl.classList.remove("hidden");
    return;
  }

  emptyEl.classList.add("hidden");
  data.forEach((siswa, i) => {
    const tr = document.createElement("tr");
    tr.innerHTML = `
      <td>${i + 1}</td>
      <td>${siswa.nama}</td>
      <td>${siswa.kelas}</td>
      <td><strong>${siswa.nilai}</strong></td>
      <td><span class="grade grade-${siswa.grade}">${siswa.grade}</span></td>
      <td><span class="status ${siswa.nilai >= NILAI_LULUS ? "lulus" : "tidak"}">
        ${siswa.nilai >= NILAI_LULUS ? "✅ Lulus" : "❌ Tidak"}
      </span></td>
      <td>
        <button class="btn-edit"  data-id="${siswa.id}">✏️</button>
        <button class="btn-hapus" data-id="${siswa.id}">🗑️</button>
      </td>
    `;
    tbody.appendChild(tr);
  });
}

function updateFilterKelas() {
  const kelasList = [...new Set(dataSiswa.map(s => s.kelas))].sort();
  const select    = document.getElementById("filterKelas");
  const current   = select.value;
  select.innerHTML = '<option value="">Semua Kelas</option>'
    + kelasList.map(k => `<option value="${k}">${k}</option>`).join("");
  select.value = current;
}

function render() {
  renderSummary();
  renderTabel();
  updateFilterKelas();
}

// ===== AKSI =====
function tambahSiswa(nama, kelas, nilai) {
  try {
    Validasi.nama(nama);
    Validasi.kelas(kelas);
    Validasi.nilai(nilai);

    const siswa = new Siswa(nama, kelas, nilai);
    dataSiswa.push(siswa);
    StorageService.simpan(dataSiswa);
    render();
    return true;
  } catch (e) {
    tampilError(e.message);
    return false;
  }
}

function hapusSiswa(id) {
  if (!confirm("Hapus siswa ini?")) return;
  dataSiswa = dataSiswa.filter(s => s.id !== Number(id));
  StorageService.simpan(dataSiswa);
  render();
}

function editSiswa(id) {
  const siswa = dataSiswa.find(s => s.id === Number(id));
  if (!siswa) return;

  const nilaBaru = prompt(`Edit nilai untuk ${siswa.nama}:`, siswa.nilai);
  if (nilaBaru === null) return;

  try {
    Validasi.nilai(nilaBaru);
    siswa.nilai = Number(nilaBaru);
    StorageService.simpan(dataSiswa);
    render();
  } catch (e) {
    tampilError(e.message);
  }
}

function exportJSON() {
  const blob = new Blob(
    [JSON.stringify(dataSiswa, null, 2)],
    { type: "application/json" }
  );
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement("a");
  a.href     = url;
  a.download = "data-siswa.json";
  a.click();
  URL.revokeObjectURL(url);
}
