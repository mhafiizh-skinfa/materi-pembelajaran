Step 6.1 — Apa itu OOP?

Konsep dasar pemrograman berorientasi objek

📌 Definisi

OOP = cara menulis program dengan mengelompokkan data & fungsi ke dalam sebuah "cetakan" yang bisa dibuat berkali-kali.


🍪 Analogi: OOP seperti Cetakan Kue

  • Cetakan Kue = CLASS (blueprint / template)
  • Kue jadi      = OBJECT (hasil dari class)
  • Satu cetakan bisa buat BANYAK kue → satu class bisa buat BANYAK object
Pilar Arti Analogi
Encapsulation Sembunyikan data internal Kapsul obat — isi tersembunyi
Inheritance Class anak mewarisi class induk Anak mewarisi sifat orang tua
Polymorphism Satu method, beda perilaku Hewan bersuara berbeda-beda
Abstraction Sembunyikan detail kompleks Setir mobil — tak perlu tahu mesinnya
Step 6.2 — Class (ES6+)

📌 Anatomi Class

  • class → keyword untuk membuat class
  • constructor() → dijalankan otomatis saat new NamaClass()
  • this → merujuk ke object yang sedang dibuat
  • Method → function di dalam class
📄 javascript 
class Siswa {

  // Constructor → dijalankan otomatis saat "new Siswa(...)"
  constructor(nama, kelas, nilai) {
    this.nama  = nama;
    this.kelas = kelas;
    this.nilai = nilai;
  }

  // Method biasa
  sapa() {
    return `Halo, saya ${this.nama} dari kelas ${this.kelas}`;
  }

  getStatus() {
    return this.nilai >= 75
      ? "✅ Lulus"
      : "❌ Tidak Lulus";
  }

  getGrade() {
    if      (this.nilai >= 90) return "A";
    else if (this.nilai >= 80) return "B";
    else if (this.nilai >= 70) return "C";
    else if (this.nilai >= 60) return "D";
    else                       return "E";
  }
}

// Membuat object (instance)
const siswa1 = new Siswa("Budi",  "XII-A", 85);
const siswa2 = new Siswa("Ani",   "XII-B", 92);

console.log(siswa1.sapa());      // Halo, saya Budi dari kelas XII-A
console.log(siswa2.getGrade()); // A
console.log(siswa1 instanceof Siswa); // true
Step 6.3 — Getter & Setter
📄 javascript 
class Produk {
  constructor(nama, harga, stok) {
    this.nama   = nama;
    this._harga = harga; // "_" = konvensi semi-private
    this._stok  = stok;
  }

  // GETTER → akses seperti property, bukan method
  get harga() {
    return `Rp${this._harga.toLocaleString("id-ID")}`;
  }

  get info() {
    return `${this.nama} | ${this.harga} | Stok: ${this._stok}`;
  }

  // SETTER → validasi sebelum mengubah nilai
  set harga(nilaiHarga) {
    if (nilaiHarga < 0) {
      console.log("❌ Harga tidak boleh negatif!");
      return;
    }
    this._harga = nilaiHarga;
  }

  set stok(jumlah) {
    if (jumlah < 0) {
      console.log("❌ Stok tidak boleh negatif!");
      return;
    }
    this._stok = jumlah;
  }
}

const laptop = new Produk("Laptop ASUS", 8500000, 10);
console.log(laptop.harga);  // Rp8.500.000
console.log(laptop.info);   // Laptop ASUS | Rp8.500.000 | Stok: 10

laptop.harga = 9000000;     // ✅
laptop.stok  = -5;          // ❌ Stok tidak boleh negatif!
Step 6.4 — Inheritance (Pewarisan)

📌 Konsep

  • extends → keyword untuk mewarisi class parent
  • super() → memanggil constructor class parent
  • Child class bisa override method dari parent
  • Child class bisa tambah property & method baru
📄 javascript 
// CLASS INDUK (Parent)
class Hewan {
  constructor(nama, suara, kaki) {
    this.nama  = nama;
    this.suara = suara;
    this.kaki  = kaki;
  }
  bersuara() {
    return `${this.nama} berkata: "${this.suara}!"`;
  }
  makan(makanan) {
    return `${this.nama} sedang makan ${makanan} 🍽️`;
  }
}

// CLASS ANAK (Child)
class Kucing extends Hewan {
  constructor(nama, warna) {
    super(nama, "Meow", 4); // panggil constructor parent
    this.warna = warna;
  }
  menggulungBadan() {
    return `${this.nama} sedang menggulung badan 🐱`;
  }
  // Override method dari parent
  bersuara() {
    return `${super.bersuara()} ...lembut sekali!`;
  }
}

class Anjing extends Hewan {
  constructor(nama, ras) {
    super(nama, "Woof", 4);
    this.ras = ras;
  }
  jaga() {
    return `${this.nama} (${this.ras}) sedang menjaga rumah 🐕`;
  }
}

const kucing = new Kucing("Kitty", "oranye");
const anjing = new Anjing("Rex", "Herder");

console.log(kucing.bersuara());       // Kitty berkata: "Meow!" ...lembut sekali!
console.log(kucing.makan("ikan"));   // dari parent ✅
console.log(anjing.jaga());          // Rex (Herder) sedang menjaga rumah 🐕
console.log(kucing instanceof Hewan); // true
Step 6.5 — Encapsulation (Private Field)

📌 Konsep

  • Private field diawali dengan #
  • Tidak bisa diakses dari luar class → lebih aman!
  • Gunakan public method untuk akses data private
📄 javascript 
class RekeningBank {
  // Private fields
  #saldo;
  #pin;
  #riwayat = [];

  constructor(pemilik, saldoAwal, pin) {
    this.pemilik = pemilik; // public
    this.#saldo  = saldoAwal; // private
    this.#pin    = pin;       // private
  }

  #validasiPin(pin) {
    return this.#pin === pin;
  }

  setor(jumlah, pin) {
    if (!this.#validasiPin(pin)) return "❌ PIN salah!";
    if (jumlah <= 0)              return "❌ Jumlah tidak valid!";
    this.#saldo += jumlah;
    return `✅ Setor Rp${jumlah.toLocaleString("id-ID")} berhasil`;
  }

  tarik(jumlah, pin) {
    if (!this.#validasiPin(pin))  return "❌ PIN salah!";
    if (jumlah > this.#saldo)     return "❌ Saldo tidak cukup!";
    this.#saldo -= jumlah;
    return `✅ Tarik Rp${jumlah.toLocaleString("id-ID")} berhasil`;
  }

  cekSaldo(pin) {
    if (!this.#validasiPin(pin)) return "❌ PIN salah!";
    return `💰 Saldo: Rp${this.#saldo.toLocaleString("id-ID")}`;
  }
}

const rek = new RekeningBank("Budi", 1000000, "1234");
console.log(rek.cekSaldo("1234"));      // 💰 Saldo: Rp1.000.000
console.log(rek.setor(500000, "1234")); // ✅ Setor berhasil
console.log(rek.tarik(200000, "0000")); // ❌ PIN salah!
// console.log(rek.#saldo); // ❌ SyntaxError!
Step 6.6 — Polymorphism
📄 javascript 
class Bentuk {
  hitungLuas()     { return 0; }
  hitungKeliling() { return 0; }
  info() {
    return `Luas: ${this.hitungLuas().toFixed(2)} | Keliling: ${this.hitungKeliling().toFixed(2)}`;
  }
}

class Persegi extends Bentuk {
  constructor(sisi) { super(); this.sisi = sisi; }
  hitungLuas()     { return this.sisi ** 2; }
  hitungKeliling() { return 4 * this.sisi; }
}

class Lingkaran extends Bentuk {
  constructor(r) { super(); this.r = r; }
  hitungLuas()     { return Math.PI * this.r ** 2; }
  hitungKeliling() { return 2 * Math.PI * this.r; }
}

const bentukList = [
  new Persegi(5),
  new Lingkaran(7),
];

// Polymorphism → method sama, hasil beda!
bentukList.forEach(b => console.log(b.info()));
// Persegi  → Luas: 25.00  | Keliling: 20.00
// Lingkaran → Luas: 153.94 | Keliling: 43.98
