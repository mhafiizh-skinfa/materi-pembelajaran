Step 7.1 — Synchronous vs Asynchronous

Memahami perbedaan alur eksekusi kode JavaScript

📌 Definisi

  • Synchronous → kode dijalankan berurutan, satu per satu. Harus tunggu yang sebelumnya selesai dulu.
  • Asynchronous → kode bisa dijalankan tanpa menunggu. Lanjut ke baris berikutnya sambil menunggu proses selesai.

🐢 Synchronous — Antrian Kasir

Orang ke-2 HARUS tunggu
orang ke-1 selesai dulu.

Satu per satu → lambat!

🚀 Asynchronous — Restoran Nomor Antrian

Kamu pesan → dapat nomor → duduk.
Orang lain bisa pesan juga.

Makanan datang → baru diproses ✅

📄 Sync vs Async 
// ===== SYNCHRONOUS =====
console.log("1. Mulai");
console.log("2. Proses");
console.log("3. Selesai");
// Output SELALU urut: 1 → 2 → 3


// ===== ASYNCHRONOUS =====
console.log("1. Mulai");

setTimeout(() => {
  console.log("2. Proses (butuh 2 detik)");
}, 2000);

console.log("3. Selesai");
// Output:
// 1. Mulai
// 3. Selesai      ← langsung lanjut!
// 2. Proses ...   ← muncul setelah 2 detik


// ===== KENAPA BUTUH ASYNC? =====
// Operasi yang butuh waktu:
// ✅ Ambil data dari API  → fetch()
// ✅ Baca / tulis file    → fs.readFile()
// ✅ Query ke database    → db.query()
// ✅ Timer / delay        → setTimeout()
// Jika SYNC → browser FREEZE menunggu! ❌
// Jika ASYNC → tetap berjalan lancar ✅
Step 7.2 — Callback

📌 Konsep

Callback = function yang dikirim sebagai argumen ke function lain, dan dipanggil setelah operasi selesai.

📄 Callback Dasar 
// Callback sederhana
function sapa(nama, callback) {
  console.log(`Halo, ${nama}!`);
  callback(nama);
}

sapa("Budi", (nama) => {
  console.log(`Selamat datang, ${nama}!`);
});
// Halo, Budi!
// Selamat datang, Budi!


// Simulasi Async dengan setTimeout
function ambilData(callback) {
  console.log("⏳ Mengambil data...");
  setTimeout(() => {
    const data = { id: 1, nama: "Budi", nilai: 85 };
    callback(null, data); // null = tidak ada error
  }, 2000);
}

ambilData((error, data) => {
  if (error) {
    console.log("❌ Error:", error);
    return;
  }
  console.log("✅ Data diterima:", data);
});
console.log("⚡ Kode lain tetap jalan...");
// ⏳ Mengambil data...
// ⚡ Kode lain tetap jalan...
// ✅ Data diterima: { id:1, nama:'Budi', nilai:85 }
❌

Callback Hell — jika callback terlalu bersarang, kode jadi susah dibaca (Pyramid of Doom). Solusinya → gunakan Promise!

❌ Callback Hell 
// Makin dalam makin susah dibaca!
ambilUser(1, (err, user) => {
  ambilPost(user.id, (err, post) => {
    ambilKomentar(post.id, (err, komentar) => {
      simpanLog(komentar, (err, log) => {
        // 😵 Pyramid of Doom!
        // semakin dalam = semakin susah maintain
      });
    });
  });
});
Step 7.3 — Promise

📌 Konsep — 3 Status Promise

  • ⏳ Pending    → masih dalam proses
  • ✅ Fulfilled → berhasil (resolve dipanggil)
  • ❌ Rejected  → gagal (reject dipanggil)
Method Fungsi Kapan Jalan
.then() Tangkap hasil sukses Saat Promise fulfilled ✅
.catch() Tangkap error Saat Promise rejected ❌
.finally() Selalu dijalankan Baik sukses maupun gagal
Promise.all() Jalankan semua paralel Tunggu semua selesai
Promise.allSettled() Jalankan semua paralel Tidak peduli ada yang gagal
Promise.race() Ambil yang tercepat Selesai pertama kali
📄 Membuat & Menggunakan Promise 
// Membuat Promise
function ambilSiswa(id) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      const db = [
        { id: 1, nama: "Budi",  nilai: 85 },
        { id: 2, nama: "Ani",   nilai: 92 },
        { id: 3, nama: "Citra", nilai: 67 },
      ];
      const siswa = db.find(s => s.id === id);

      if (siswa) resolve(siswa);
      else       reject(new Error(`Siswa id ${id} tidak ditemukan`));
    }, 1000);
  });
}

// Menggunakan Promise
ambilSiswa(2)
  .then(siswa  => console.log("✅", siswa.nama, siswa.nilai))
  .catch(err   => console.log("❌", err.message))
  .finally(()  => console.log("🏁 Selesai!"));


// Promise.all → paralel, tunggu semua selesai
Promise.all([ambilSiswa(1), ambilSiswa(2), ambilSiswa(3)])
  .then(([s1, s2, s3]) => console.log(s1.nama, s2.nama, s3.nama))
  .catch(err => console.log("❌ Salah satu gagal:", err.message));


// Promise.allSettled → tidak peduli ada yang gagal
Promise.allSettled([ambilSiswa(1), ambilSiswa(99)])
  .then(results => {
    results.forEach((r, i) => {
      if (r.status === "fulfilled")
        console.log(`✅ ${i}:`, r.value.nama);
      else
        console.log(`❌ ${i}:`, r.reason.message);
    });
  });
📄 Promise Chaining 
// Chaining → jauh lebih rapi dari callback hell! ✅
ambilUser(1)
  .then(user => {
    console.log("👤 User:", user.nama);
    return ambilPost(user.id);    // return promise baru
  })
  .then(post => {
    console.log("📝 Post:", post.judul);
    return ambilKomentar(post.id); // return promise baru
  })
  .then(komentar => {
    console.log("💬 Komentar:", komentar);
  })
  .catch(err => {
    console.log("❌ Error:", err.message); // tangkap SEMUA error
  })
  .finally(() => console.log("🏁 Selesai!"));
Step 7.4 — Async / Await ✅

📌 Konsep

  • async → menandai function sebagai asynchronous
  • await → tunggu Promise selesai, baru lanjut ke baris berikutnya
  • await hanya bisa dipakai di dalam async function
  • Kode terlihat seperti synchronous, tapi tetap async!

😵 Promise Chaining

ambilUser(1)
  .then(user => {
    return ambilPost(user.id);
  })
  .then(post => {
    return ambilKomentar(post.id);
  })
  .then(komentar => {
    console.log(komentar);
  })
  .catch(err => {...})

✅ Async / Await

async function getData() {
  try {
    const user     = await ambilUser(1);
    const post     = await ambilPost(user.id);
    const komentar = await ambilKomentar(post.id);
    console.log(komentar);
  } catch(err) {...}
}
📄 Async/Await Lengkap 
// Fungsi helper delay
function delay(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// Async/Await dengan try-catch
async function tampilSiswa(id) {
  try {
    console.log(`⏳ Mencari siswa id: ${id}...`);
    const siswa = await ambilSiswa(id); // tunggu sampai selesai
    console.log(`✅ ${siswa.nama} | Nilai: ${siswa.nilai}`);
  } catch (error) {
    console.log(`❌ ${error.message}`);
  } finally {
    console.log("🏁 Pencarian selesai!\n");
  }
}

tampilSiswa(1);  // ✅ Budi | Nilai: 85
tampilSiswa(99); // ❌ Siswa id 99 tidak ditemukan


// ❌ Sequential (lambat) → nunggu satu-satu
async function lambat() {
  const a = await ambilSiswa(1); // tunggu 1 detik
  const b = await ambilSiswa(2); // tunggu 1 detik lagi
  const c = await ambilSiswa(3); // tunggu 1 detik lagi → total 3 detik!
}

// ✅ Parallel (cepat) → jalankan bersamaan
async function cepat() {
  const [a, b, c] = await Promise.all([
    ambilSiswa(1),
    ambilSiswa(2),
    ambilSiswa(3),
  ]); // total hanya ~1 detik! 🚀
  console.log(a.nama, b.nama, c.nama);
}


// Fetch API dengan Async/Await
async function ambilDataUser() {
  try {
    const response = await fetch("https://jsonplaceholder.typicode.com/users/1");

    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }

    const user = await response.json();
    console.log("✅ Nama  :", user.name);
    console.log("✅ Email :", user.email);
    console.log("✅ Kota  :", user.address.city);

  } catch (error) {
    console.log("❌ Gagal:", error.message);
  }
}

ambilDataUser();
