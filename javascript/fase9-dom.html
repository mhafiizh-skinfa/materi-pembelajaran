Step 9.1 — Apa itu DOM?

Document Object Model — representasi HTML sebagai object

📌 Definisi

  • DOM = Document Object Model
  • Representasi halaman HTML sebagai struktur pohon (tree) dari object
  • JavaScript bisa membaca, mengubah, menambah, dan menghapus elemen HTML melalui DOM
  • Entry point: document → object utama untuk mengakses DOM
📄 document
└── 🌐 <html>
├── 🗂️ <head>
│   ├── <title> "Belajar JS"
│   └── <meta>
└── 📦 <body>
    ├── <h1> id="judul"
    ├── <div> class="container"
    │   ├── <p>
    │   └── <button>
    └── <script>

📄 addEventListener 
const tombol = document.querySelector("#tombol");

// Cara 1: addEventListener ✅ (DIREKOMENDASIKAN)
tombol.addEventListener("click", function(event) {
  console.log("Diklik!");
  console.log(event.target);    // elemen yang diklik
  console.log(event.type);     // "click"
});

// Cara 2: Arrow function
tombol.addEventListener("click", (e) => {
  e.preventDefault();   // cegah perilaku default (misal: submit form)
  e.stopPropagation(); // hentikan event bubbling
});

// ===== EVENT OBJECT =====
document.addEventListener("keydown", (e) => {
  console.log(e.key);        // "Enter", "a", "ArrowUp", dll
  console.log(e.code);       // "KeyA", "Enter", dll
  console.log(e.ctrlKey);    // true jika Ctrl ditekan
  console.log(e.shiftKey);   // true jika Shift ditekan
  if (e.key === "Enter") console.log("Enter ditekan!");
});

// ===== EVENT DELEGATION =====
// Pasang 1 event listener di parent → tangkap semua child
const list = document.querySelector("#daftarSiswa");

list.addEventListener("click", (e) => {
  if (e.target.matches(".btn-hapus")) {
    e.target.closest("li").remove();
  }
  if (e.target.matches(".btn-edit")) {
    const nama = e.target.closest("li").querySelector(".nama").textContent;
    console.log("Edit:", nama);
  }
});

// ===== HAPUS EVENT LISTENER =====
function handleKlik() { console.log("Klik!"); }
tombol.addEventListener("click", handleKlik);
tombol.removeEventListener("click", handleKlik); // hapus listener

📄 localStorage & sessionStorage 
// ===== localStorage =====
// Simpan string
localStorage.setItem("nama", "Budi");
localStorage.setItem("umur", "17");

// Simpan object (harus di-JSON.stringify dulu)
const siswa = { nama: "Budi", kelas: "XII", nilai: 85 };
localStorage.setItem("siswa", JSON.stringify(siswa));

// Baca data
const nama = localStorage.getItem("nama"); // "Budi"
const dataObj = JSON.parse(localStorage.getItem("siswa"));

// Hapus data
localStorage.removeItem("nama");     // hapus satu
localStorage.clear();              // hapus semua

// Cek ada/tidak
const ada = localStorage.getItem("nama") !== null;


// ===== Helper functions untuk localStorage =====
const storage = {
  simpan(key, value) {
    localStorage.setItem(key, JSON.stringify(value));
  },
  ambil(key, defaultVal = null) {
    const data = localStorage.getItem(key);
    return data ? JSON.parse(data) : defaultVal;
  },
  hapus(key) { localStorage.removeItem(key); },
  bersih()   { localStorage.clear(); }
};

storage.simpan("user", { nama: "Budi", role: "admin" });
const user = storage.ambil("user");
console.log(user.nama); // Budi

location.hostname;          // "localhost"

console.log(params.get("nama"));  // "budi"


// ===== HISTORY =====
